def find_invalid_ids(range)
  result = []
  a, b = range
  if a.length.odd?
    a = next_invalid_id(a)
    if a > b
      return result
    end
  end
end

def next_invalid_id(number)
  sequence = 10 ** ((number.length + 1) / 2 - 1)
  sequence.to_s * 2
end

def part_1(input)
  password, dial = 0, START
  input.each do |rotation|
    direction = rotation[0] == "R" ? 1 : -1
    distance = rotation[1..]&.to_i
    dial = (dial + (direction * distance)) % LENGTH
    if dial == 0
      password += 1
    end
  end
  password
end

def part_2(input)
  password, dial = 0, START
  input.each do |step|
    direction = step[0] == "R" ? 1 : -1
    distance = step[1..]&.to_i
    if distance > LENGTH
      password += distance / LENGTH
      distance = distance - (distance / LENGTH) * LENGTH
    end
    if dial == 0
      delta = 100
      password += 1
    else
      delta = direction == 1 ? LENGTH - dial : dial
    end
    if distance > delta
      password += 1
    end
    dial = (dial + (direction * distance)) % LENGTH
  end
  password
end
